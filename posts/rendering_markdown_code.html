Rendering markdown code

January 20, 2025

In the process of making this blog, I had a whole bunch of markdown files I needed to render.

In these markdown files, there are sometimes code snippets, which look boring without syntax highlighting.

To make these look better I implemented this syntax highlighting library

  • remark-rehype
  • starry-night
  • rehype-starry-night

How these are implemented here is demonstrated in the lib/posts/posts.js file of the github repository for this site:

js
1import path from 'path';
2import matter from 'gray-matter';
3import rehypeStringify from 'rehype-stringify';
4import remarkRehype from 'remark-rehype';
5import rehypeStarryNight from "@microflash/rehype-starry-night";
6import remarkParse from "remark-parse";
7import { unified } from 'unified';
8import remarkInlineCodeLang from "@microflash/rehype-starry-night/remark-inline-code-lang";
9import rehypeStarryNightInline from "@microflash/rehype-starry-night/rehype-starry-night-inline";
10import fileManager from './files';
11
12export async function getPostData(id) {
13  const fullPath = path.join(postsDirectory, `${id}.md`);
14  const fileContents = fileManager.getFile(fullPath);
15  const matterResult = matter(fileContents);
16
17  const processedContent = await unified()
18    .use(remarkParse)
19    .use(remarkInlineCodeLang)
20    .use(remarkRehype, { allowDangerousHtml: true })
21    .use(rehypeStarryNight)
22    .use(rehypeStarryNightInline)
23    .use(rehypeStringify, { allowDangerousHtml: true })
24    .process(matterResult.content);
25
26  const contentHtml = processedContent.toString();
27
28  return {
29    id,
30    contentHtml,
31    ...matterResult.data,
32  };
33}
