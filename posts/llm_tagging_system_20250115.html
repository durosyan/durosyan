llm tagging system

January 15, 2025

I designed this script to be able to quickly create and semantically tag blog content from the command line.

The content itself is not ai generated, only the tags.

Usage:

bash
./new_entry.sh 'blog title'

Script:

bash
1#!/bin/bash
2if [[ $# -gt 1 ]]; then
3    echo "Error: Title should be a single argument. If it contains spaces, enclose it in quotes."
4    exit 1
5fi
6
7if [[ -z $1 ]]; then
8    read -p "Enter the title of your journal entry: " title
9else
10    title=$1
11fi
12
13if [[ -f .env ]]; then
14    export $(grep -v '^#' .env | xargs)
15else
16    echo ".env file not found. Exiting..."
17    exit 1
18fi
19
20if [[ -z $OPEN_AI_KEY ]]; then
21    echo "OPEN_AI_KEY is not set in the .env file. Exiting..."
22    exit 1
23fi
24
25ai_prompt="Create a series of topic tags from the following title, reply with only the comma separated tags in the format 'tag, tag, tag' no additional information or text, at least one of these tags must be used (Ai, Analysis, Economy, Engineering, Hacking, Philosophy, Research, Snippets, Tinkering). The title is: "$title
26response=$(curl https://api.openai.com/v1/chat/completions \
27-H "Content-Type: application/json" \
28-H "Authorization: Bearer $OPEN_AI_KEY" \
29-d '{
30    "model": "gpt-4o",
31    "store": true,
32    "messages": [
33        {"role": "user", "content": "'"$ai_prompt"'"}
34    ]
35    }' | jq -r '.choices[0].message.content')
36
37if [[ -z $response ]]; then
38    echo "Error calling openai. Exiting..."
39    exit 1
40fi
41
42# Parse tags from the response of the AI (lowercase all, tags are separated by commas)
43tags=$(echo "$response" | tr '[:upper:]' '[:lower:]' | tr -d '\"' | tr -d "'")
44echo "Tags: $tags"
45
46if [[ ! -z $title ]]; then
47    filetitle=$(echo "$title" | tr '[:upper:]' '[:lower:]' | tr ' ' '_')
48    filename=$filetitle"_$(date +%Y%m%d)"
49    cat << EOF > $filename.md
50---
51title: '$title'
52date: '$(date +"%Y-%m-%d")'
53tags: '$tags'
54---
55EOF
56    echo "Created new blog entry: $filename.md"
57else
58    echo "No title entered. Exiting..."
59    exit 1
60fi
